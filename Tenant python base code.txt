import requests
import urllib3
urllib3.disable_warnings(urllib3.exceptions.InsecureRequestWarning)

base_url = "https://35.200.176.139"
login_url = f"{base_url}/login"
tenant_url = f"{base_url}/api/tenant"

username = "hiring-2"
password = "hiring-2"

session = requests.Session()

login_payload = {
    "username": username,
    "password": password
}

login_response = session.post(
    login_url,
    json=login_payload,
    verify=False
)

if login_response.status_code != 200:
    print("Login failed:", login_response.text)
    exit()

avi_sessionid = session.cookies.get("avi-sessionid")
csrftoken = session.cookies.get("csrftoken")

if not avi_sessionid or not csrftoken:
    print("Failed to get session or CSRF token.")
    exit()

print("Login successful.")
print("Session ID:", avi_sessionid)
print("CSRF Token:", csrftoken)

headers = {
    "Content-Type": "application/json",
    "X-CSRFToken": csrftoken,
    "Referer": base_url
}

tenant_payload = {
    "name": "J. Hariharan"
}

create_response = session.post(
    tenant_url,
    json=tenant_payload,
    headers=headers,
    verify=False
)

if create_response.status_code == 200:
    print("Tenant created successfully.")
    print(create_response.json())
else:
    print("Failed to create tenant:", create_response.status_code)
    print(create_response.text)
